\section*{Problem 13:}
Let $\bold{u} = \begin{bmatrix} 1\\ 2 \\ 3\end{bmatrix}$, 
	$\bold{v} = \begin{bmatrix}-1\\ 2 \\ 4\end{bmatrix}$, 
	$\bold{w} = \begin{bmatrix} 0\\ 1 \\-1\end{bmatrix}$,
	$c = -2$, and $d =3$.
\bigskip
\bigskip
\newline
Compute:
\begin{enumerate}[a)]
\item $\bold{u}+\bold{v}$
\newline
$
	\begin{bmatrix} 1\\ 2 \\ 3\end{bmatrix} +
	\begin{bmatrix}-1\\ 2 \\ 4\end{bmatrix} = 
	\begin{bmatrix} 0\\ 4 \\ 7\end{bmatrix}
$
\newline
\item $c\bold{u}+d\bold{w}$
\newline
$
	-2\begin{bmatrix} 1\\ 2 \\ 3\end{bmatrix} +
	3\begin{bmatrix} 0\\ 1 \\-1\end{bmatrix} = 
	\begin{bmatrix} 2\\ 4 \\ 6\end{bmatrix} +
	\begin{bmatrix} 0\\ 3 \\-3\end{bmatrix} =
	\begin{bmatrix} 2\\ 7 \\3\end{bmatrix}
$
\newline
\item $\bold{u}+\bold{v}+\bold{w}$
\newline
$
	\begin{bmatrix} 1\\ 2 \\ 3\end{bmatrix} +
	\begin{bmatrix}-1\\ 2 \\ 4\end{bmatrix} +
	\begin{bmatrix} 0\\ 1 \\-1\end{bmatrix} =
	\begin{bmatrix} 0\\ 5 \\6\end{bmatrix}
$
\newline
\item $c\bold{u}+d\bold{v}+\bold{w}$
\newline
$
	-2\begin{bmatrix} 1\\ 2 \\ 3\end{bmatrix} +
	3\begin{bmatrix}-1\\ 2 \\ 4\end{bmatrix} +
	\begin{bmatrix} 0\\ 1 \\-1\end{bmatrix} =
	\begin{bmatrix} -2\\ -4 \\ -6\end{bmatrix} +
	\begin{bmatrix}-3\\ 6 \\ 12\end{bmatrix} +
	\begin{bmatrix} 0\\ 1 \\-1\end{bmatrix} =
	\begin{bmatrix} -5\\ 3 \\5\end{bmatrix}
$
\end{enumerate}
